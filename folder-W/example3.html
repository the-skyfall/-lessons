Урок 2: Продвинутые техники CSS

1. Flexbox

<div class="flex-container">
        <div class="flex-item">Элемент 1</div>
        <div class="flex-item">Элемент 2</div>
        <div class="flex-item">Элемент 3</div>
    </div>
              <style>
                    /* Контейнер Flexbox с выравниванием элементов по горизонтали */
                    .flex-container {
                        display: flex; /* Включаем Flexbox */
                        justify-content: space-between; /* Распределяем элементы с промежутками между ними */
                        align-items: center; /* Выравниваем элементы по центру вертикально */
                        padding: 20px; /* Отступ внутри контейнера */
                        background-color: #f0f0f0; /* Цвет фона контейнера */
                        border: 1px solid #ddd; /* Граница контейнера */
                        border-radius: 5px; /* Скругляем углы контейнера */
                    }
    
                    /* Стили для отдельных элементов Flexbox */
                    .flex-item {
                        background-color: #007bff; /* Цвет фона элементов */
                        color: #fff; /* Цвет текста элементов */
                        padding: 20px; /* Отступ внутри элементов */
                        border-radius: 5px; /* Скругляем углы элементов */
                        flex: 1; /* Растягиваем элементы равномерно */
                        margin: 5px; /* Отступы вокруг элементов */
                        text-align: center; /* Выравниваем текст по центру */
                    }
    
                    /* Изменяем размер второго элемента Flexbox */
                    .flex-item:nth-child(2) {
                        flex: 2; /* Увеличиваем ширину второго элемента в два раза по сравнению с остальными */
                    }
                </style>

2. CSS Grid

<div class="grid-container">
        <div class="grid-item">Элемент 1</div>
        <div class="grid-item">Элемент 2</div>
        <div class="grid-item">Элемент 3</div>
        <div class="grid-item">Элемент 4</div>
        <div class="grid-item">Элемент 5</div>
    </div>
              <style>
                    /* Контейнер CSS Grid с колонками и отступами */
                    .grid-container {
                        display: grid; /* Включаем CSS Grid */
                        grid-template-columns: repeat(3, 1fr); /* Создаем три равные колонки */
                        gap: 10px; /* Отступы между элементами сетки */
                        padding: 20px; /* Отступ внутри контейнера */
                        background-color: #e0e0e0; /* Цвет фона контейнера */
                        border: 1px solid #ddd; /* Граница контейнера */
                        border-radius: 5px; /* Скругляем углы контейнера */
                    }
    
                    /* Стили для элементов CSS Grid */
                    .grid-item {
                        background-color: #007bff; /* Цвет фона элементов */
                        color: #fff; /* Цвет текста элементов */
                        padding: 20px; /* Отступ внутри элементов */
                        border-radius: 5px; /* Скругляем углы элементов */
                        text-align: center; /* Выравниваем текст по центру */
                    }
    
                    /* Изменяем размер второго элемента Grid */
                    .grid-item:nth-child(2) {
                        grid-column: span 2; /* Занимаем две колонки */
                    }
                </style>

3. Трансформации и Переходы

<div class="transform-example">
        Трансформируем
    </div>
              <style>
                    /* Элемент с трансформациями и переходами */
                    .transform-example {
                        width: 100px; /* Ширина элемента */
                        height: 100px; /* Высота элемента */
                        background-color: #007bff; /* Цвет фона элемента */
                        color: #fff; /* Цвет текста элемента */
                        display: flex; /* Включаем Flexbox для центрирования текста */
                        align-items: center; /* Выравниваем текст по вертикали */
                        justify-content: center; /* Выравниваем текст по горизонтали */
                        border-radius: 10px; /* Скругляем углы элемента */
                        transition: transform 0.3s ease, background-color 0.3s ease; /* Переходы для трансформации и изменения фона */
                        margin-bottom: 20px; /* Отступ снизу элемента */
                    }
    
                    /* Эффект при наведении */
                    .transform-example:hover {
                        transform: scale(1.2) rotate(15deg); /* Увеличиваем размер и поворачиваем элемент */
                        background-color: #28a745; /* Меняем цвет фона элемента */
                    }
                </style>

4. CSS Фильтры

<div class="filter-example"></div>
              <style>
                    /* Элемент с применением CSS фильтров */
                    .filter-example {
                        width: 300px; /* Ширина элемента */
                        height: 200px; /* Высота элемента */
                        background-image: url('https://via.placeholder.com/300x200'); /* Фоновое изображение */
                        background-size: cover; /* Заполняем элемент фоновым изображением */
                        filter: grayscale(100%) blur(5px); /* Применяем черно-белый фильтр и размываем изображение */
                        transition: filter 0.3s ease; /* Переход для фильтров */
                        margin-bottom: 20px; /* Отступ снизу элемента */
                    }
    
                    /* Эффект при наведении */
                    .filter-example:hover {
                        filter: grayscale(0%) blur(0); /* Убираем фильтры при наведении */
                    }
                </style>

5. Псевдоклассы и Псевдоэлементы

<div class="pseudo-class-example">
        Пример псевдокласса
    </div>
              <style>
                    /* Элемент с псевдоклассами и псевдоэлементами */
                    .pseudo-class-example {
                        padding: 20px; /* Отступ внутри элемента */
                        background-color: #007bff; /* Цвет фона элемента */
                        color: #fff; /* Цвет текста элемента */
                        border-radius: 5px; /* Скругляем углы элемента */
                        position: relative; /* Устанавливаем позиционирование для псевдоэлементов */
                        margin-bottom: 20px; /* Отступ снизу элемента */
                    }
    
                    /* Псевдоэлемент до контента */
                    .pseudo-class-example::before {
                        content: '★'; /* Добавляем звезду перед контентом */
                        color: #ffd700; /* Цвет звезды */
                        position: absolute; /* Абсолютное позиционирование */
                        top: 10px; /* Располагаем на 10 пикселей ниже верхнего края */
                        left: 10px; /* Располагаем на 10 пикселей левее левого края */
                    }
    
                    /* Эффект при наведении */
                    .pseudo-class-example:hover {
                        background-color: #28a745; /* Меняем цвет фона элемента при наведении */
                    }
    
                    /* Псевдоэлемент после контента */
                    .pseudo-class-example::after {
                        content: ' - Наведите для изменения'; /* Добавляем текст после контента */
                        color: #fff; /* Цвет текста */
                        font-style: italic; /* Курсивное начертание */
                    }
                </style>
